49bool SDCard::begin() {
50 if (_initialized) {
51 return true;
52 }
53
54 // Initialize SPI bus
55 initSPI();
56
57 // Try to initialize SD card
58 if (!SD.begin(SD_CS, *_spi)) {
59 Serial.println("[SDCard] Initialization failed!");
60 _initialized = false;
61 return false;
62 }
63
64 // Check card type
65 uint8_t cardType = SD.cardType();
66 if (cardType == CARD_NONE) {
67 Serial.println("[SDCard] No SD card attached");
68 _initialized = false;
69 return false;
70 }
71
72 _initialized = true;
73
74 // Print card info
75 Serial.println("[SDCard] Initialization successful");
76 Serial.print("[SDCard] Type: ");
77 Serial.println(getCardType());
78 Serial.print("[SDCard] Size: ");
79 Serial.print(getCardSizeMB());
80 Serial.println(" MB");
81 Serial.print("[SDCard] Free: ");
82 Serial.print(getFreeSpaceMB());
83 Serial.println(" MB");
84
85 return true;
86}

247void SDCard::listDirectory(const char* path, uint8_t indent) {
248 if (!_initialized) return;
249
250 File dir = SD.open(path);
251 if (!dir) {
252 Serial.print("[SDCard] Failed to open directory: ");
253 Serial.println(path);
254 return;
255 }
256
257 if (!dir.isDirectory()) {
258 Serial.println("[SDCard] Not a directory");
259 dir.close();
260 return;
261 }
262
263 File file = dir.openNextFile();
264 while (file) {
265 for (uint8_t i = 0; i < indent; i++) {
266 Serial.print(" ");
267 }
268
269 Serial.print(file.name());
270
271 if (file.isDirectory()) {
272 Serial.println("/");
273 // Recursively list subdirectories
274 String subPath = String(path) + "/" + String(file.name());
275 listDirectory(subPath.c_str(), indent + 1);
276 } else {
277 Serial.print(" - ");
278 Serial.print(file.size());
279 Serial.println(" bytes");
280 }
281
282 file.close();
283 file = dir.openNextFile();
284 }
285
286 dir.close();
287}
