182void Core0_NVSValidation(void* parameter) {
183 LOG("[Core 0] NVS Validation Task Started");
184
185 // Simulate some validation steps with status updates
186 updateStatus("Checking NVS...");
187 delay(500);
188
189 // Step 1: Initialize NVS
190 updateStatus("Initializing NVS...");
191 if (!using_nvs_eeprom.begin()) {
192 g_nvsResult.status = NVSStatus::INITIALIZATION_FAILED;
193 updateStatus("NVS Init Failed!");
194 validationComplete = true;
195 vTaskDelete(NULL);
196 return;
197 }
198 LOG("[Core 0] NVS Initialized");
199 delay(500);
200
201 // Step 2: Validate entire NVS
202 updateStatus("Validating data...");
203 g_nvsResult = using_nvs_eeprom.validateNVS();
204 delay(500);
205
206 // Step 3: Report results
207 LOG("=== Core 0 NVS Validation Results ===");
208 LOG("Status: ");
209
210 switch (g_nvsResult.status) {
211 case NVSStatus::VALID:
212 LOG("VALID");
213 updateStatus("Validation: PASSED");
214 break;
215 case NVSStatus::UPGRADED:
216 LOG("UPGRADED");
217 updateStatus("Validation: UPGRADED");
218 break;
219 case NVSStatus::INVALID_VERSION:
220 LOG("INVALID_VERSION");
221 updateStatus("Validation: BAD VERSION");
222 break;
223 case NVSStatus::MISSING_ZAPNUMBER:
224 LOG("MISSING_ZAPNUMBER");
225 updateStatus("Setup Required");
226 break;
227 case NVSStatus::MISSING_PASSWORD:
228 LOG("MISSING_PASSWORD");
229 updateStatus("Password Required");
230 break;
231 case NVSStatus::CRC_FAILED:
232 LOG("CRC_FAILED - TAMPERING DETECTED!");
233 updateStatus("SECURITY ALERT!");
234 break;
235 case NVSStatus::INITIALIZATION_FAILED:
236 LOG("INITIALIZATION_FAILED");
237 updateStatus("Hardware Error");
238 break;
239 default:
240 LOG("NOT_CHECKED");
241 updateStatus("Unknown Status");
242 break;
243 }
244
245 LOGF("Version: Current=%d, Expected=%d",
246 g_nvsResult.currentVersion,
247 g_nvsResult.expectedVersion);
248 LOGF("ZapNumber: Valid=%d, Value=%s",
249 g_nvsResult.zapNumberValid,
250 g_nvsResult.zapNumber);
251 LOGF("Password: Valid=%d", g_nvsResult.passwordHashValid);
252 LOGF("CRC: Valid=%d, Value=0x%08X",
253 g_nvsResult.crcValid,
254 g_nvsResult.calculatedCRC);
255 LOGF("Upgraded: %d", g_nvsResult.wasUpgraded);
256 LOG("====================================");
257
258 delay(1000); // Show final status for 1 second
259
260 // Mark validation as complete
261 validationComplete = true;
262
263 LOG("[Core 0] Validation Complete - Task Ending");
264
265 // Task complete, delete itself
266 vTaskDelete(NULL);
267}

337void Core1_DisplayHandler(void* parameter) {
338 LOG("[Core 1] Display Handler Task Started");
339
340 while (true) {
341 // Handle LVGL timer
342 if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
343 lv_timer_handler();
344 xSemaphoreGive(displayMutex);
345 }
346
347 // Check if validation is complete
348 if (validationComplete) {
349 // Hide spinner
350 if (spinner != NULL) {
351 lv_obj_del(spinner);
352 spinner = NULL;
353 }
354
355 // Run loader logic
356 core0_loaderLogic();
357
358 // End this task as validation phase is done
359 LOG("[Core 1] Display task ending - validation complete");
360 vTaskDelete(NULL);
361 return;
362 }
363
364 delay(5);
365 }
366}

376void setup() {
377 // Initialize Serial for debugging (keep for emergency fallback)
378 Serial.begin(115200);
379 delay(1000);
380
381 // Initialize Logger first
382 init_logger();
383
384 LOG("\n\n=== EARS Dual-Core System Starting ===");
385 LOGF("Setup running on Core: %d", xPortGetCoreID());
386
387 // Create mutex for display synchronization
388 displayMutex = xSemaphoreCreateMutex();
389
390 // Step 1: Initialize backlight
391 pinMode(GFX_BL, OUTPUT);
392 digitalWrite(GFX_BL, HIGH);
393 LOG("Backlight: ON");
394
395 // Step 2: Initialize display
396 LOG("Initializing display...");
397 gfx->begin();
398 gfx->fillScreen(EARS_RGB565_BLACK);
399 LOG("Display: Initialized");
400
401 // Step 3: Initialize LVGL
402 LOG("Initializing LVGL...");
403 lv_init();
404 lv_tick_set_cb(millis_cb);
405
406 disp = lv_display_create(screenWidth, screenHeight);
407 lv_display_set_flush_cb(disp, my_disp_flush);
408 lv_display_set_buffers(disp, buf1, buf2, sizeof(buf1), LV_DISPLAY_RENDER_MODE_PARTIAL);
409 LOG("LVGL: Initialized");
410
411 // Step 4: Create loading screen with spinner
412 LOG("Creating loading screen...");
413 lv_obj_t *scr = lv_screen_active();
414 lv_obj_set_style_bg_color(scr, lv_color_hex(EARS_RGB888_BLACK), 0);
415
416 // Create spinner (loading animation)
417 spinner = lv_spinner_create(scr);
418 lv_obj_set_size(spinner, 80, 80);
419 lv_obj_center(spinner);
420 lv_obj_set_style_arc_color(spinner, lv_color_hex(0x00FF00), LV_PART_INDICATOR);
421
422 // Create status label below spinner
423 status_label = lv_label_create(scr);
424 lv_label_set_text(status_label, "Starting...");
425 lv_obj_set_style_text_color(status_label, lv_color_hex(EARS_RGB888_WHITE), 0);
426 lv_obj_set_style_text_font(status_label, &lv_font_montserrat_16, 0);
427 lv_obj_align(status_label, LV_ALIGN_CENTER, 0, 60);
428
429 LOG("Loading screen: Created");
430
431 // Step 5: Create Core 0 task for NVS validation
432 LOG("Creating Core 0 validation task...");
433 xTaskCreatePinnedToCore(
434 Core0_NVSValidation, // Task function
435 "NVS_Validation", // Name
436 4096, // Stack size (bytes)
437 NULL, // Parameters
438 1, // Priority (1 = low, higher = more priority)
439 &Core0_ValidationTask, // Task handle
440 0 // Core 0
441 );
442
443 // Step 6: Create Core 1 task for display handling
444 LOG("Creating Core 1 display task...");
445 xTaskCreatePinnedToCore(
446 Core1_DisplayHandler, // Task function
447 "Display_Handler", // Name
448 8192, // Stack size (bytes) - larger for LVGL
449 NULL, // Parameters
450 2, // Priority (higher than validation)
451 &Core1_DisplayTask, // Task handle
452 1 // Core 1
453 );
454
455 LOG("=== Setup Complete ===");
456 LOG("Core 0: Running NVS validation");
457 LOG("Core 1: Running display animation\n");
458}
