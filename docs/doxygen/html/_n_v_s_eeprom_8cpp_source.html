95uint32_t NVSEeprom::calculateCRC32(const uint8_t* data, size_t length) {
96 uint32_t crc = 0xFFFFFFFF;
97
98 for (size_t i = 0; i < length; i++) {
99 crc ^= data[i];
100 for (uint8_t j = 0; j < 8; j++) {
101 crc = (crc >> 1) ^ (0xEDB88320 & -(crc & 1));
102 }
103 }
104
105 return ~crc;
106}

183bool NVSEeprom::isValidZapNumber(const String& zapNumber) {
184 // Must be exactly 6 characters
185 if (zapNumber.length() != 6) {
186 return false;
187 }
188
189 // First two characters must be letters (A-Z)
190 if (!isAlpha(zapNumber[0]) || !isAlpha(zapNumber[1])) {
191 return false;
192 }
193
194 // Last four characters must be digits (0-9)
195 for (int i = 2; i < 6; i++) {
196 if (!isDigit(zapNumber[i])) {
197 return false;
198 }
199 }
200
201 return true;
202}

314NVSValidationResult NVSEeprom::validateNVS() {
315 NVSValidationResult result;
316 result.expectedVersion = CURRENT_VERSION;
317
318 // Step 1: Check NVS initialization
319 if (!Preferences::begin(NAMESPACE, true)) {
320 result.status = NVSStatus::INITIALIZATION_FAILED;
321 end();
322 return result;
323 }
324
325 // Step 2: Get and check version
326 result.currentVersion = getUShort(KEY_VERSION, 0);
327
328 // Check if upgrade is needed
329 if (result.currentVersion < CURRENT_VERSION) {
330 end(); // Close read-only session
331
332 // Attempt upgrade
333 if (upgradeNVS(result.currentVersion, CURRENT_VERSION)) {
334 result.wasUpgraded = true;
335 result.currentVersion = CURRENT_VERSION;
336 } else {
337 result.status = NVSStatus::INVALID_VERSION;
338 return result;
339 }
340
341 // Reopen for continued validation
342 if (!Preferences::begin(NAMESPACE, true)) {
343 result.status = NVSStatus::INITIALIZATION_FAILED;
344 return result;
345 }
346 } else if (result.currentVersion > CURRENT_VERSION) {
347 // Version from future - cannot handle
348 result.status = NVSStatus::INVALID_VERSION;
349 end();
350 return result;
351 }
352
353 // Step 3: Check ZapNumber
354 String zapNum = getString(KEY_ZAPNUMBER, "");
355 if (zapNum.length() == 0 || !isValidZapNumber(zapNum)) {
356 result.zapNumberValid = false;
357 result.status = NVSStatus::MISSING_ZAPNUMBER;
358 end();
359 return result;
360 }
361 result.zapNumberValid = true;
362 zapNum.toCharArray(result.zapNumber, 7);
363
364 // Step 4: Check password hash
365 String pwdHash = getString(KEY_PASSWORD_HASH, "");
366 if (pwdHash.length() == 0) {
367 result.passwordHashValid = false;
368 result.status = NVSStatus::MISSING_PASSWORD;
369 end();
370 return result;
371 }
372 result.passwordHashValid = true;
373
374 // Step 5: Check overall CRC32
375 uint32_t storedCRC = getUInt(KEY_NVS_CRC, 0);
376 end(); // Close before calculating new CRC
377
378 uint32_t calculatedCRC = calculateNVSCRC();
379 result.calculatedCRC = calculatedCRC;
380
381 if (storedCRC != calculatedCRC) {
382 result.crcValid = false;
383 result.status = NVSStatus::CRC_FAILED;
384 return result;
385 }
386 result.crcValid = true;
387
388 // Step 6: All checks passed
389 if (result.wasUpgraded) {
390 result.status = NVSStatus::UPGRADED;
391 } else {
392 result.status = NVSStatus::VALID;
393 }
394
395 return result;
396}
