89uint32_t NVSEeprom::calculateCRC32(const uint8_t* data, size_t length) {
90 uint32_t crc = 0xFFFFFFFF;
91
92 for (size_t i = 0; i < length; i++) {
93 crc ^= data[i];
94 for (uint8_t j = 0; j < 8; j++) {
95 crc = (crc >> 1) ^ (0xEDB88320 & -(crc & 1));
96 }
97 }
98
99 return ~crc;
100}
