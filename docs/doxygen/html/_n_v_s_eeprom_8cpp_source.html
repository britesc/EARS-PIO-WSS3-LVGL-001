94uint32_t NVSEeprom::calculateCRC32(const uint8_t* data, size_t length) {
95 uint32_t crc = 0xFFFFFFFF;
96
97 for (size_t i = 0; i < length; i++) {
98 crc ^= data[i];
99 for (uint8_t j = 0; j < 8; j++) {
100 crc = (crc >> 1) ^ (0xEDB88320 & -(crc & 1));
101 }
102 }
103
104 return ~crc;
105}

182bool NVSEeprom::isValidZapNumber(const String& zapNumber) {
183 // Must be exactly 6 characters
184 if (zapNumber.length() != 6) {
185 return false;
186 }
187
188 // First two characters must be letters (A-Z)
189 if (!isAlpha(zapNumber[0]) || !isAlpha(zapNumber[1])) {
190 return false;
191 }
192
193 // Last four characters must be digits (0-9)
194 for (int i = 2; i < 6; i++) {
195 if (!isDigit(zapNumber[i])) {
196 return false;
197 }
198 }
199
200 return true;
201}

313NVSValidationResult NVSEeprom::validateNVS() {
314 NVSValidationResult result;
315 result.expectedVersion = CURRENT_VERSION;
316
317 // Step 1: Check NVS initialization
318 if (!Preferences::begin(NAMESPACE, true)) {
319 result.status = NVSStatus::INITIALIZATION_FAILED;
320 end();
321 return result;
322 }
323
324 // Step 2: Get and check version
325 result.currentVersion = getUShort(KEY_VERSION, 0);
326
327 // Check if upgrade is needed
328 if (result.currentVersion < CURRENT_VERSION) {
329 end(); // Close read-only session
330
331 // Attempt upgrade
332 if (upgradeNVS(result.currentVersion, CURRENT_VERSION)) {
333 result.wasUpgraded = true;
334 result.currentVersion = CURRENT_VERSION;
335 } else {
336 result.status = NVSStatus::INVALID_VERSION;
337 return result;
338 }
339
340 // Reopen for continued validation
341 if (!Preferences::begin(NAMESPACE, true)) {
342 result.status = NVSStatus::INITIALIZATION_FAILED;
343 return result;
344 }
345 } else if (result.currentVersion > CURRENT_VERSION) {
346 // Version from future - cannot handle
347 result.status = NVSStatus::INVALID_VERSION;
348 end();
349 return result;
350 }
351
352 // Step 3: Check ZapNumber
353 String zapNum = getString(KEY_ZAPNUMBER, "");
354 if (zapNum.length() == 0 || !isValidZapNumber(zapNum)) {
355 result.zapNumberValid = false;
356 result.status = NVSStatus::MISSING_ZAPNUMBER;
357 end();
358 return result;
359 }
360 result.zapNumberValid = true;
361 zapNum.toCharArray(result.zapNumber, 7);
362
363 // Step 4: Check password hash
364 String pwdHash = getString(KEY_PASSWORD_HASH, "");
365 if (pwdHash.length() == 0) {
366 result.passwordHashValid = false;
367 result.status = NVSStatus::MISSING_PASSWORD;
368 end();
369 return result;
370 }
371 result.passwordHashValid = true;
372
373 // Step 5: Check overall CRC32
374 uint32_t storedCRC = getUInt(KEY_NVS_CRC, 0);
375 end(); // Close before calculating new CRC
376
377 uint32_t calculatedCRC = calculateNVSCRC();
378 result.calculatedCRC = calculatedCRC;
379
380 if (storedCRC != calculatedCRC) {
381 result.crcValid = false;
382 result.status = NVSStatus::CRC_FAILED;
383 return result;
384 }
385 result.crcValid = true;
386
387 // Step 6: All checks passed
388 if (result.wasUpgraded) {
389 result.status = NVSStatus::UPGRADED;
390 } else {
391 result.status = NVSStatus::VALID;
392 }
393
394 return result;
395}
