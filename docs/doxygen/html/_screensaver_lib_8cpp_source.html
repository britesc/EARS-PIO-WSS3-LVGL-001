127void ScreensaverLib::update() {
128 if (!_settings.enabled) return;
129 if (_settings.timeout_seconds == 0) return; // Disabled via timeout
130
131 uint32_t elapsed = millis() - _last_activity_ms;
132 uint32_t timeout_ms = _settings.timeout_seconds * 1000;
133
134 if (!_is_active && elapsed >= timeout_ms) {
135 activate();
136 }
137
138 if (_is_active) {
139 updateAnimation();
140 }
141}
