49bool Logger::begin(const char* logFilePath, const char* configFilePath, SDCard* sdCard) {
50 if (_initialized) {
51 return true;
52 }
53
54 if (!sdCard || !sdCard->isAvailable()) {
55 return false;
56 }
57
58 _sdCard = sdCard;
59 _logFilePath = String(logFilePath);
60 _configFilePath = String(configFilePath);
61
62 // Ensure log directory exists
63 int lastSlash = _logFilePath.lastIndexOf('/');
64 if (lastSlash > 0) {
65 String logDir = _logFilePath.substring(0, lastSlash);
66 if (!_sdCard->directoryExists(logDir.c_str())) {
67 _sdCard->createDirectory(logDir.c_str());
68 }
69 }
70
71 // Ensure config directory exists
72 lastSlash = _configFilePath.lastIndexOf('/');
73 if (lastSlash > 0) {
74 String configDir = _configFilePath.substring(0, lastSlash);
75 if (!_sdCard->directoryExists(configDir.c_str())) {
76 _sdCard->createDirectory(configDir.c_str());
77 }
78 }
79
80 // Load configuration (creates default if not exists)
81 loadConfig();
82
83 _initialized = true;
84
85 // Log initialization
86 info("=== Logger v2.1 Initialized ===");
87 infof("Log file: %s", _logFilePath.c_str());
88 infof("Config file: %s", _configFilePath.c_str());
89 infof("Log level: %s", getLogLevelString().c_str());
90 infof("Max file size: %d bytes (%.2f MB)", _config.maxFileSizeBytes, _config.maxFileSizeBytes / 1048576.0);
91 infof("Max rotated files: %d", _config.maxRotatedFiles);
92
93 return true;
94}

272String Logger::getTimestamp() const {
273 struct timeval tv;
274 gettimeofday(&tv, NULL);
275
276 struct tm timeinfo;
277 localtime_r(&tv.tv_sec, &timeinfo);
278
279 char buffer[32];
280 snprintf(buffer, sizeof(buffer), "%04d-%02d-%02d %02d:%02d:%02d",
281 timeinfo.tm_year + 1900,
282 timeinfo.tm_mon + 1,
283 timeinfo.tm_mday,
284 timeinfo.tm_hour,
285 timeinfo.tm_min,
286 timeinfo.tm_sec);
287
288 return String(buffer);
289}

334LogLevel Logger::parseLevelString(const String& levelStr) const {
335 String upper = levelStr;
336 upper.toUpperCase();
337
338 if (upper == "NONE") return LogLevel::NONE;
339 if (upper == "ERROR") return LogLevel::ERROR;
340 if (upper == "WARN" || upper == "WARNING") return LogLevel::WARN;
341 if (upper == "INFO") return LogLevel::INFO;
342 if (upper == "DEBUG") return LogLevel::DEBUG;
343
344 return LogLevel::NONE; // Default to NONE if invalid
345}

406bool Logger::loadConfig() {
407 JsonDocument doc;
408
409 if (!loadUnifiedConfig(doc)) {
410 // Config doesn't exist, create default
411 doc["system"]["version"] = "1.0.0";
412 doc["system"]["zap_number"] = ""; // Will be populated from NVS
413 doc["system"]["device_name"] = "EARS";
414
415 doc["logger"]["log_level"] = "DEBUG";
416 doc["logger"]["max_file_size_bytes"] = 1048576;
417 doc["logger"]["max_rotated_files"] = 3;
418
419 saveUnifiedConfig(doc);
420 }
421
422 // Extract logger section
423 JsonObject loggerObj = doc["logger"];
424 if (loggerObj.isNull()) {
425 return false;
426 }
427
428 String levelStr = loggerObj["log_level"] | "DEBUG";
429 _config.currentLevel = parseLevelString(levelStr);
430 _config.maxFileSizeBytes = loggerObj["max_file_size_bytes"] | 1048576;
431 _config.maxRotatedFiles = loggerObj["max_rotated_files"] | 3;
432
433 return true;
434}

521bool Logger::performRotation() {
522 if (!_initialized) {
523 return false;
524 }
525
526 info("Starting log rotation...");
527
528 // Delete oldest rotated file if it exists
529 String oldestFile = _logFilePath + "." + String(_config.maxRotatedFiles);
530 if (_sdCard->fileExists(oldestFile.c_str())) {
531 _sdCard->removeFile(oldestFile.c_str());
532 }
533
534 // Shift all rotated files up by one
535 for (int i = _config.maxRotatedFiles - 1; i >= 1; i--) {
536 String oldName = _logFilePath + "." + String(i);
537 String newName = _logFilePath + "." + String(i + 1);
538
539 if (_sdCard->fileExists(oldName.c_str())) {
540 // Read old file
541 String content = _sdCard->readFile(oldName.c_str());
542 // Write to new file
543 _sdCard->writeFile(newName.c_str(), content);
544 // Delete old file
545 _sdCard->removeFile(oldName.c_str());
546 }
547 }
548
549 // Rename current log to .1
550 String rotatedName = _logFilePath + ".1";
551 if (_sdCard->fileExists(_logFilePath.c_str())) {
552 String content = _sdCard->readFile(_logFilePath.c_str());
553 _sdCard->writeFile(rotatedName.c_str(), content);
554 _sdCard->removeFile(_logFilePath.c_str());
555 }
556
557 info("Log rotation completed");
558
559 return true;
560}
